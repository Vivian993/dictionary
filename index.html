    // Storage Management
    function loadData() {
      const stored = localStorage.getItem('kanghan_5a_dict_data');
      if (stored) {
        try {
          appData = JSON.parse(stored);
        } catch(e) {
          appData = DEFAULT_DATA;
        }
      } else {
        appData = DEFAULT_DATA;
        saveData();
      }

      localStorage.setItem('kanghan_5a_dict_data', JSON.stringify(appData));

    // Filtering Engine
    function getFilteredData() {
      const lessonVal = document.getElementById('lessonSelect').value;
      const searchVal = document.getElementById('searchInput').value.trim().toLowerCase();

      document.getElementById('clearSearchBtn').classList.toggle('hidden', searchVal === '');

      return appData.filter(item => {
        const matchLesson = (lessonVal === 'all') || (item.lessonId.toString() === lessonVal);
        
        if (!searchVal) return matchLesson;

        const matchChar = item.char.includes(searchVal);
        const matchZhuyin = item.zhuyin.includes(searchVal);
        const matchRadical = item.radical.includes(searchVal);
        const matchCharMeaning = item.charMeaning && item.charMeaning.includes(searchVal);
        const matchTerms = item.terms && item.terms.some(t => t.term.includes(searchVal) || (t.def && t.def.includes(searchVal)));

        return matchLesson && (matchChar || matchZhuyin || matchRadical || matchCharMeaning || matchTerms);
      });
    }

    }

    // TTS Web Speech Synthesis
    function speakText(text) {
      if (!('speechSynthesis' in window)) {
        alert("您的瀏覽器不支援語音朗讀功能！");
        return;
      }
      window.speechSynthesis.cancel(); // Stop current speech
      const rate = parseFloat(document.getElementById('voiceRate').value) || 0.9;
      const utterance = new SpeechSynthesisUtterance(text);
      utterance.lang = 'zh-TW';
      utterance.rate = rate;
      window.speechSynthesis.speak(utterance);
    }

    function playAllQuizAudio() {
      const filtered = getFilteredData();
      let allTerms = [];
      filtered.forEach(item => item.terms.forEach(t => allTerms.push(t.term)));

      if (allTerms.length === 0) return;

      let index = 0;
      function speakNext() {
        if (index >= allTerms.length) return;
        const term = allTerms[index];
        const utterance = new SpeechSynthesisUtterance(term);
        utterance.lang = 'zh-TW';
        utterance.rate = parseFloat(document.getElementById('voiceRate').value) || 0.9;
        utterance.onend = () => {
          index++;
          setTimeout(speakNext, 2000); // Wait 2s between terms for dictation
        };
        window.speechSynthesis.speak(utterance);
      }
      speakNext();
    }

    // Canvas Image Compression Handler
    function handleImageCompress(e) {
      const file = e.target.files[0];
      if (!file) return;

      const reader = new FileReader();
      reader.onload = function(event) {
        const img = new Image();
        img.src = event.target.result;
        img.onload = function() {
          const canvas = document.createElement('canvas');
          const maxDim = 300; // Resize to max width/height 300px for space efficiency
          let width = img.width;
          let height = img.height;

          if (width > height) {
            if (width > maxDim) {
              height *= maxDim / width;
              width = maxDim;
            }
          } else {
            if (height > maxDim) {
              width *= maxDim / height;
              height = maxDim;
            }
          }

          canvas.width = width;
          canvas.height = height;
          const ctx = canvas.getContext('2d');
          ctx.drawImage(img, 0, 0, width, height);

          // Compress image to JPEG format with quality 0.6
          const compressedDataUrl = canvas.toDataURL('image/jpeg', 0.6);
          document.getElementById('editImageData').value = compressedDataUrl;
          
          // Preview
          document.getElementById('imagePreview').src = compressedDataUrl;
          document.getElementById('imagePreviewContainer').classList.remove('hidden');
        };
      };
      reader.readAsDataURL(file);
    }

    // Form Handle & Save
    function handleSaveForm(e) {
      e.preventDefault();

      const editId = document.getElementById('editCharId').value;
      const lessonId = parseInt(document.getElementById('editLesson').value, 10);
      const char = document.getElementById('editChar').value.trim();
      const zhuyin = document.getElementById('editZhuyin').value.trim();
      const radical = document.getElementById('editRadical').value.trim();
      const charMeaning = document.getElementById('editCharMeaning').value.trim();

      const term1 = document.getElementById('editTerm1').value.trim();
      const term1Zhuyin = document.getElementById('editTerm1Zhuyin').value.trim();
      const term1Def = document.getElementById('editTerm1Def').value.trim();

      const term2 = document.getElementById('editTerm2').value.trim();
      const term2Zhuyin = document.getElementById('editTerm2Zhuyin').value.trim();
      const term2Def = document.getElementById('editTerm2Def').value.trim();

      const image = document.getElementById('editImageData').value;

      const terms = [
        { term: term1, zhuyin: term1Zhuyin, def: term1Def },
        { term: term2, zhuyin: term2Zhuyin, def: term2Def }
      ];

      if (editId) {
        // Edit existing
        const idx = appData.findIndex(item => item.id === editId);
        if (idx !== -1) {
          appData[idx] = { id: editId, lessonId, char, zhuyin, radical, charMeaning, terms, image };
        }
      } else {
        // Add new
        const newObj = {
          id: 'char_' + Date.now(),
          lessonId,
          char,
          zhuyin,
          radical,
          charMeaning,
          terms,
          image
        };
        appData.push(newObj);
      }

      saveData();
      resetForm();
      alert("儲存成功！");
      switchTab('dict');
    }

    function editCharItem(id) {
      const item = appData.find(i => i.id === id);
      if (!item) return;

      switchTab('manage');

      document.getElementById('editCharId').value = item.id;
      document.getElementById('editLesson').value = item.lessonId;
      document.getElementById('editChar').value = item.char;
      document.getElementById('editZhuyin').value = item.zhuyin;
      document.getElementById('editRadical').value = item.radical;
      document.getElementById('editCharMeaning').value = item.charMeaning || '';

      if (item.terms[0]) {
        document.getElementById('editTerm1').value = item.terms[0].term || '';
        document.getElementById('editTerm1Zhuyin').value = item.terms[0].zhuyin || '';
        document.getElementById('editTerm1Def').value = item.terms[0].def || '';
      }

      if (item.terms[1]) {
        document.getElementById('editTerm2').value = item.terms[1].term || '';
        document.getElementById('editTerm2Zhuyin').value = item.terms[1].zhuyin || '';
        document.getElementById('editTerm2Def').value = item.terms[1].def || '';
      }

      if (item.image) {
        document.getElementById('editImageData').value = item.image;
        document.getElementById('imagePreview').src = item.image;
        document.getElementById('imagePreviewContainer').classList.remove('hidden');
      } else {
        removeImage();
      }
    }

    function importData(e) {
      const file = e.target.files[0];
      if (!file) return;

      const reader = new FileReader();
      reader.onload = function(evt) {
        try {
          const imported = JSON.parse(evt.target.result);
          if (Array.isArray(imported)) {
            appData = imported;
            saveData();
            renderApp();
            alert("備份資料匯入成功！");
          } else {
            alert("格式不符合！");
          }
        } catch(err) {
          alert("無法讀取 JSON 備份檔案。");
        }
      };
      reader.readAsText(file);
    }
