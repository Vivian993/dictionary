    function loadData() {
      const stored = localStorage.getItem('kanghan_5a_pink_data_v3');
      if (stored) {
        try { appData = JSON.parse(stored); } catch(e) { appData = DEFAULT_DATA; }
      } else {

      }

    function populateLessonDropdowns() {
      const select = document.getElementById('lessonSelect');

    function switchTab(tab) {
      currentTab = tab;
      ['dict', 'quiz', 'manage'].forEach(t => {
        document.getElementById(`view-${t}`).classList.toggle('hidden', t !== tab);
        document.getElementById(`tab-${t}`).classList.toggle('active', t === tab);
      });
      renderApp();
    }

    /* 🌐 完全免費的「現查現有」萌典資料庫連線 */
    async function lookupMoeDict() {
      const charInput = document.getElementById('editChar').value.trim();
      if (!charInput) return alert("請先在第一格輸入您想查的字！");
      
      const btn = document.querySelector("#editForm button[onclick='lookupMoeDict()']");
      const originalText = btn.innerHTML;
      btn.innerHTML = `查尋中...`;
      btn.style.pointerEvents = 'none';

      try {
        // 使用教育部萌典公開 API (完全免費，不需金鑰)
        const response = await fetch(`https://www.moedict.tw/a/${encodeURIComponent(charInput)}.json`);
        if (!response.ok) throw new Error('找不到');
        
        const data = await response.json();
        const hetero = data.heteronyms[0]; // 抓取第一個讀音資料
        
        if (hetero) {
          // 自動填入注音 (移除特殊符號)
          if (hetero.bopomofo) {
            document.getElementById('editZhuyin').value = hetero.bopomofo.replace(/<[^>]*>?/gm, '').trim();
          }
          // 自動填入部首與筆劃
          if (data.radical) document.getElementById('editRadical').value = data.radical + "部";
          if (data.stroke_count) document.getElementById('editStrokes').value = data.stroke_count + "畫";
          
          // 自動填入字義解釋 (抓取第一個解釋並移除標籤)
          if (hetero.definitions && hetero.definitions.length > 0) {
            let defText = hetero.definitions[0].def;
            defText = defText.replace(/<[^>]*>?/gm, '').replace(/~/g, '').replace(/`|'/g, '');
            document.getElementById('editCharMeaning').value = defText;
          }

          // 預設將該字複製到圈詞1，方便填寫
          document.getElementById('editTerm1').value = charInput;
          document.getElementById('editTerm1Zhuyin').value = document.getElementById('editZhuyin').value;
          document.getElementById('editTerm1Def').value = document.getElementById('editCharMeaning').value;

          alert("✅ 成功找到字詞資料，已自動填入各欄位！\n您可以稍微檢查修改後，按下「儲存進我的字典」。");
        }
      } catch (error) {
        alert(`無法自動取得「${charInput}」的資料。這可能是一個罕見字或詞語。\n即將為您直接開啟教育部辭典網頁讓您查詢！`);
        window.open(`https://dict.revised.moe.edu.tw/search.jsp?md=1&word=${encodeURIComponent(charInput)}`, '_blank');
      } finally {
        btn.innerHTML = originalText;
        btn.style.pointerEvents = 'auto';
      }
    }

    function speakText(text) {
      if (!('speechSynthesis' in window)) return;
      window.speechSynthesis.cancel();
      const u = new SpeechSynthesisUtterance(text);
      u.lang = 'zh-TW';
      u.rate = 0.85;
      window.speechSynthesis.speak(u);
    }

    function renderApp() {
      const lessonVal = document.getElementById('lessonSelect').value;
      const searchVal = document.getElementById('searchInput').value.trim().toLowerCase();

      const filtered = appData.filter(item => {
        const matchLesson = (lessonVal === 'all') || (item.lessonId.toString() === lessonVal);
        const matchChar = item.char.includes(searchVal) || item.zhuyin.includes(searchVal) || item.terms.some(t => t.term.includes(searchVal));
        return matchLesson && (!searchVal || matchChar);
      });

      if (currentTab === 'dict') renderDict(filtered);
      else if (currentTab === 'quiz') renderQuiz(filtered);
    }

        return;

      `).join('');
    }

    function renderQuiz(items) {
      const container = document.getElementById('quizListContainer');
      let allTerms = [];
      items.forEach(item => item.terms.forEach(t => allTerms.push({ ...t, lessonId: item.lessonId })));
      
      if(allTerms.length === 0){
        container.innerHTML = '<div style="text-align:center; padding:2rem; color:#9d174d;">🌸 本課尚無測驗內容！</div>';
        return;
      }

      container.innerHTML = allTerms.map((t, idx) => `
        <div class="quiz-card">
          <div style="flex:1;">
            <div style="font-size:0.75rem; color:#9d174d;">第 ${t.lessonId} 課 圈詞 ${idx + 1}</div>
            <div style="font-size:1.05rem; font-weight:800;">
              <span class="quiz-mask ${allRevealed ? 'revealed' : ''}" onclick="this.classList.toggle('revealed')">${t.term}</span>
              <span style="font-size:0.85rem; color:#db2777;">(${t.zhuyin})</span>
            </div>
          </div>
          <button onclick="speakText('${t.term}')" class="btn btn-pink">🔊 發音</button>
        </div>
      `).join('');
    }

      document.querySelectorAll('.quiz-mask').forEach(m => m.classList.toggle('revealed', allRevealed));
    }

    function exportData() {
      const d = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(appData, null, 2));
      const a = document.createElement('a'); a.href = d; a.download = "國語字典備份.json"; document.body.appendChild(a); a.click(); a.remove();
    }

    function triggerImport() { document.getElementById('importFile').click(); }
    
    function handleFileImport(e) {
      const f = e.target.files[0]; if (!f) return;
      const r = new FileReader();
      r.onload = function(x) { try { appData = JSON.parse(x.target.result); saveData(); renderApp(); alert("匯入成功！"); } catch(y) {} }; r.readAsText(f);
    }

    function handleSaveForm(e) {
      e.preventDefault();
      const id = document.getElementById('editCharId').value;
      const data = {
        id: id || ('char_' + Date.now()),
        lessonId: parseInt(document.getElementById('editLesson').value, 10),
        char: document.getElementById('editChar').value.trim(),
        zhuyin: document.getElementById('editZhuyin').value.trim(),
        radical: document.getElementById('editRadical').value.trim(),
        strokes: document.getElementById('editStrokes').value.trim(),
        charMeaning: document.getElementById('editCharMeaning').value.trim(),
        terms: [{ term: document.getElementById('editTerm1').value.trim(), zhuyin: document.getElementById('editTerm1Zhuyin').value.trim(), def: document.getElementById('editTerm1Def').value.trim(), sentence: document.getElementById('editTerm1Sentence').value.trim() }]
      };
      if (id) { const i = appData.findIndex(x => x.id === id); if (i !== -1) appData[i] = data; }
      else appData.push(data);
      saveData();
      document.getElementById('editForm').reset(); switchTab('dict');
    }

    function editCharItem(id) {
      const item = appData.find(x => x.id === id); if (!item) return;
      switchTab('manage');
      document.getElementById('editCharId').value = item.id;
      document.getElementById('editLesson').value = item.lessonId;
      document.getElementById('editChar').value = item.char;
      document.getElementById('editZhuyin').value = item.zhuyin;
      document.getElementById('editRadical').value = item.radical;
      document.getElementById('editStrokes').value = item.strokes;
      document.getElementById('editCharMeaning').value = item.charMeaning;
      if (item.terms[0]) {
        document.getElementById('editTerm1').value = item.terms[0].term;
        document.getElementById('editTerm1Zhuyin').value = item.terms[0].zhuyin;
        document.getElementById('editTerm1Def').value = item.terms[0].def;
        document.getElementById('editTerm1Sentence').value = item.terms[0].sentence;
      }
    }

    function deleteCharItem(id) { if (confirm("確定刪除此字詞？")) { appData = appData.filter(x => x.id !== id); saveData(); renderApp(); } }
